Aim: Construct a program that demonstrates implementation of Inheritance - Single and Multilevel

Practice Exercise 9: Write a program to calculate heat transfer by conduction implementing single inheritance.

Theory:

The base class Material is created to store thermal conductivity. The derived class Conduction computes the heat transfer rate using Fourier's Law, which is dependent on the material's thermal conductivity, area, temperature difference, and length.

Algorithm:

  1. Define a class Material with a member variable k for thermal conductivity.
  2. Define a derived class Conduction that inherits from Material. This class should have the following members:
  3. A: Area through which heat is transferred.
  4. dT: Temperature difference.
  5. L: Length of the material.
  6. In the Conduction class, implement a method heatTransfer() that calculates the heat transfer rate using the formula: [ Q = \frac{k \times A \times \Delta T}{L} ]
  7. In the main function, create an object of Conduction and set values for k, A, dT, and L.
  8. Call the heatTransfer() method and display the result.

Code:

#include <iostream>
using namespace std;

class Material {
protected:
    float k; // Thermal conductivity
public:
    Material(float conductivity) { k = conductivity; }
};

class Conduction : public Material {
    float A, dT, L;
public:
    Conduction(float conductivity, float area, float tempDiff, float length)
        : Material(conductivity) {
        A = area; dT = tempDiff; L = length;
    }

    float heatTransfer() { return (k * A * dT) / L; }
};

int main() {
    Conduction obj(205, 1.5, 100, 0.05);
    cout << "Heat Transfer Rate: " << obj.heatTransfer() << " W" << endl;
    return 0;
}

Practice Exercise 10: Write a program to calculate heat transfer by forced convection using multilevel inheritance.

Theory:

The base class Fluid holds the convective heat transfer coefficient h. The derived class Convection calculates heat transfer for convection, and the class ForcedConvection further extends Convection to calculate heat transfer in forced convection, including a velocity factor.

Algorithm:

Code:

#include <iostream>
using namespace std;

class Fluid {
protected:
    float h; // Convective heat transfer coefficient
public:
    Fluid(float coefficient) { h = coefficient; }
};

class Convection : public Fluid {
protected:
    float A, dT;
public:
    Convection(float coefficient, float area, float tempDiff)
        : Fluid(coefficient) {
        A = area; dT = tempDiff;
    }

    float heatTransfer() { return h * A * dT; }
};

class ForcedConvection : public Convection {
    float velocity;
public:
    ForcedConvection(float coefficient, float area, float tempDiff, float vel)
        : Convection(coefficient, area, tempDiff) {
        velocity = vel;
    }

    float enhancedHeatTransfer() { return heatTransfer() * velocity; }
};

int main() {
    ForcedConvection obj(50, 2.0, 30, 1.5);
    cout << "Enhanced Heat Transfer: " << obj.enhancedHeatTransfer() << " W" << endl;
    return 0;
}

Aim: Construct a program that demonstrates implementation of Inheritance - Multiple, Hierarchical and Hybrid.

Practice Exercise 11: Write a program to calculate heat transfer from conduction and radiation using multiple inheritance.

Theory:

The Conduction class calculates heat transfer by conduction, and the Radiation class calculates heat transfer by radiation. The HeatTransfer class inherits from both and computes the total heat transfer by adding the results from both methods.

Algorithm:

Code:

#include <iostream>
#include <cmath>
using namespace std;

class Conduction {
protected:
    float k, A, dT, L;
public:
    Conduction(float conductivity, float area, float tempDiff, float length) {
        k = conductivity; A = area; dT = tempDiff; L = length;
    }

    float conductionHeat() { return (k * A * dT) / L; }
};

class Radiation {
protected:
    float emissivity, sigma, T1, T2;
public:
    Radiation(float e, float area, float t1, float t2) {
        emissivity = e; sigma = 5.67e-8; T1 = t1; T2 = t2; 
    }

    float radiationHeat() { return emissivity * sigma * A * (pow(T1, 4) - pow(T2, 4)); }
};

class HeatTransfer : public Conduction, public Radiation {
public:
    HeatTransfer(float conductivity, float area, float tempDiff, float length,
                 float e, float t1, float t2)
        : Conduction(conductivity, area, tempDiff, length), Radiation(e, area, t1, t2) {}

    float totalHeat() { return conductionHeat() + radiationHeat(); }
};

int main() {
    HeatTransfer obj(205, 1.5, 100, 0.05, 0.9, 500, 300);
    cout << "Total Heat Transfer: " << obj.totalHeat() << " W" << endl;
    return 0;
}

Practice Exercise 12: Write a program to calculate heat transfer by conduction and convection using hierarchical inheritance.

Theory:

The HeatTransfer base class stores common parameters. The Conduction class and Convection class inherit from it and calculate heat transfer based on their respective methods. The program demonstrates the use of a common base class for different heat transfer modes.

Algorithm:

Code:

#include <iostream>
using namespace std;

class HeatTransfer {
protected:
    float A, dT;
public:
    HeatTransfer(float area, float tempDiff) { A = area; dT = tempDiff; }
};

class Conduction : public HeatTransfer {
    float k, L;
public:
    Conduction(float area, float tempDiff, float conductivity, float length)
        : HeatTransfer(area, tempDiff) {
        k = conductivity; L = length;
    }

    float heatTransfer() { return (k * A * dT) / L; }
};

class Convection : public HeatTransfer {
    float h;
public:
    Convection(float area, float tempDiff, float coefficient)
        : HeatTransfer(area, tempDiff) {
        h = coefficient;
    }

    float heatTransfer() { return h * A * dT; }
};

int main() {
    Conduction c(1.5, 100, 205, 0.05);
    Convection conv(2.0, 50, 30);

    cout << "Conduction Heat Transfer: " << c.heatTransfer() << " W" << endl;
    cout << "Convection Heat Transfer: " << conv.heatTransfer() << " W" << endl;

    return 0;
}
